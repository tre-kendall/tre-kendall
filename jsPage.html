This was a project I created this year in my JavaScript coding class. This code created a calendar that showed the April schedule for the Kansas City Royals's spring training. I spaced out the opponents schedule and the game location so I wouldn't have extremely long lines of code, which in return makes it easier to read.

Here is a link to the working webpage

      
// global variables
var daysOfWeek = ["Sunday", "Monday", "Tuesday", "Wednesday",
   "Thursday", "Friday", "Saturday"];
var opponents = ["(off)", "MIN", "MIN",
    "DET", "(off)", "DET", "DET",
    "SEA", "SEA", "SEA",
    "SEA", "CLE", "CLE", "CLE",
    "CWS", "CWS", "CWS", "NYY",
    "NYY", "NYY", "NYY",
    "TB", "TB", "TB", "(off)",
    "LAA", "LAA", "LAA", "TB",
    "TB", "(off)"];
var gameLocation =
   ["", "home", "home", "away", "", "away", "away",
    "home", "home", "home", "home", "home", "home", "home",
    "away", "away", "away", "away", "away", "away", "away",
    "away", "away", "away", "", "home", "home", "home",
    "home", "home", ""];

// function to place daysOfWeek values in header row cells
function addColumnHeaders() {
   var i = 0;
   while (i < 7) {
      document.getElementsByTagName("th")[i].innerHTML = daysOfWeek[i];
      i++;
   }
}

// function to place day of month value in first p element
// within each table data cell that has an id
function addCalendarDates() {
   var i = 1;
   var paragraphs = "";
   do {
      var tableCell = document.getElementById("08-" + i);
      paragraphs = tableCell.getElementsByTagName("p");
      paragraphs[0].innerHTML = i;
      i++;
   } while (i <= 30);
}

// function to place opponents and gameLocation values in
// second p element within each table data cell that has an id
function addGameInfo() {
   var paragraphs = "";
   for (var i = 0; i < 31; i++) {
      var date = i+1;
      var tableCell = document.getElementById("08-" + date);
      paragraphs = tableCell.getElementsByTagName("p");
/*      if (gameLocation[i] === "away") {
         paragraphs[1].innerHTML = "@ ";
      }
      if (gameLocation[i] === "home") {
         paragraphs[1].innerHTML = "vs ";
      }*/
/*      if (gameLocation[i] === "away") {
         paragraphs[1].innerHTML = "@ ";
      }
      else {
         if (gameLocation[i] === "home") {
            paragraphs[1].innerHTML = "vs ";
         }
      } */
      switch (gameLocation[i]) {
         case "away":
            paragraphs[1].innerHTML = "@ ";
            break;
         case "home":
            paragraphs[1].innerHTML = "vs ";
            break;
      }
      paragraphs[1].innerHTML += opponents[i];
   }
}

// function to populate calendar
function setUpPage() {
   addColumnHeaders();
   addCalendarDates();
   addGameInfo();
}

// runs setUpPage() function when page loads
if (window.addEventListener) {
   window.addEventListener("load", setUpPage, false);
} else if (window.attachEvent) {
   window.attachEvent("load", setUpPage);
}
